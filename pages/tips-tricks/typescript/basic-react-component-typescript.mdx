# Basic React component with Typescript

```jsx
interface Props {
  name: string
  isDev: boolean
  favLanguage: string
}

const Dev: React.FC<Props> = ({ name, isDev, favLanguage }) => {
  return (
    <div>
      <h1>{name}</h1>
      {isDev && (
        <h4>Favorite programming language</h4>
        <span>{favLanguage}</span>
      )}
    </div>
  )
}

export default Dev

// Using component
<Dev name="Nico Pellerin" isDev={true} favLanguage="Rust" />
```
